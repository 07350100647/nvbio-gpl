114 //
115 // Helper templated class used to automatically unroll the central DP loop for
116 // the edit distance scoring computation.
117 // The template arguments provide the beginning of the iteration and the band size.
118 // The iteration parameter REV_J runs in reverse, starting from BAND_LEN, and
119 // is used to index band BAND_LEN - REV_J.
120 // This had to be done to allow specializing the first and last steps, which
121 // correspond to the plain constants BAND_LEN and 1.
122 // Using the band index would have required specializing case 0 and BAND_LEN-1,
123 // but the latter is unfortunately not possible, as it is an expression dependent
124 // on a template parameter (a rather silly limitation of C++ - I agree).
125 //
126 template <uint32 REV_J, uint32 BAND_LEN>
127 struct EditDistCore
128 {
129  template <typename Band_type>
130  __device__ static void run(const uint8* genome_cache, const uint32 i, const uint8 g, const uint8 r, Band_type& band, uint8& lowest)
131  {
132  const uint32 J = BAND_LEN - REV_J;
133 
134  const uint32 diagnal = (genome_cache[ threadIdx.x + BLOCKDIM * ((i+J) & BAND_LEN) ] == r) ?
135  get<J>( band ) :
136  get<J>( band )+SUBSTITUTE_PENALTY;
137  const uint32 top = get<J+1>( band ) + DELETE_PENALTY;
138  const uint32 left = get<J-1>( band ) + INSERT_PENALTY;
139  get<J>( band ) = min( min( top, left ), diagnal ); // TODO: use video instructions here
140  lowest = get<J>( band ) < lowest ? get<J>( band ) : lowest;
141 
142  EditDistCore<REV_J-1,BAND_LEN>::run( genome_cache, i, g, r, band, lowest );
143  }
144 };

146 //
147 // Special case for band 0.
148 //
149 template <uint32 BAND_LEN>
150 struct EditDistCore<BAND_LEN,BAND_LEN>
151 {
152  template <typename Band_type>
153  __device__ static void run(const uint8* genome_cache, const uint32 i, const uint8 g, const uint8 r, Band_type& band, uint8& lowest)
154  {
155  const uint32 diagnal = (genome_cache[ threadIdx.x + BLOCKDIM * (i & BAND_LEN) ] == r) ?
156  get<0>( band ) :
157  get<0>( band )+SUBSTITUTE_PENALTY;
158  const uint32 top = get<1>( band ) + DELETE_PENALTY;
159  lowest = get<0>( band ) = min( top, diagnal );
160 
161  EditDistCore<BAND_LEN-1,BAND_LEN>::run( genome_cache, i, g, r, band, lowest );
162  }
163 };
164 //
165 // Special case for band BAND_LEN-1.
166 //
167 template <uint32 BAND_LEN>
168 struct EditDistCore<1u,BAND_LEN>
169 {
170  template <typename Band_type>
171  __device__ static void run(const uint8* genome_cache, const uint32 i, const uint8 g, const uint8 r, Band_type& band, uint8& lowest)

177  get<BAND_LEN-1>( band ) = min( left, diagnal );
178  lowest = get<BAND_LEN-1>( band ) < lowest ? get<BAND_LEN-1>( band ) : lowest;
