465 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
466 uint64 hash3(uint64 key)
467 {
468  uint32 hash = 0u;
469 
470  #if defined(__CUDA_ARCH__)
471  #pragma unroll
472  #endif
473  for (uint32 i = 0; i < 8; ++i)
474  {
475  hash = (hash << 4) + ((key >> (i*8)) & 255u); // shift/mix
476 
477  // get high nybble
478  const uint32 hi_bits = hash & 0xF0000000;
479  if (hi_bits != 0u)
480  hash ^= hi_bits >> 24; // xor high nybble with second nybble
481 
482  hash &= ~hi_bits; // clear high nybble
483  }
