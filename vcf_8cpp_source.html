31 // loads a VCF 4.2 file, appending the data to output
32 bool loadVCF(SNPDatabase& output, const char *file_name)
33 {
34  BufferedTextFile file(file_name);
35  char *line, *end;
36  uint32 line_counter = 0;
37 
38  while((line = file.next_record(&end)))
39  {
40  line_counter++;
41  *end = '\0';
42 
43  // strip out comments
44  char *comment = strchr(line, '#');
45  if (comment)
46  *comment = '\0';
47 
48  // skip all leading whitespace
49  while (*line == ' ' || *line == '\t' || *line == '\r')
50  {
51  line++;
52  }
53 
54  if (*line == '\0')
55  {
56  // empty line, skip
57  continue;
58  }
59 
60  // parse the entries in each record
61  char *chrom, *pos, *id, *ref, *alt, *qual, *filter;
62 
63 // ugly macro to tokenize the string based on strchr
64 #define NEXT(prev, next) \
65  { \
66  next = strchr(prev, '\t'); \
67  if (!next) { \
68  log_error(stderr, "Error parsing VCF file (line %d): incomplete variant\n", line_counter); \
69  return false; \
70  } \
71  *next = '\0'; \
72  next++; \
73  }
74 
75  chrom = line;
76  NEXT(chrom, pos);
77  NEXT(pos, id);
78  NEXT(id, ref);
79  NEXT(ref, alt);
80  NEXT(alt, qual);
81  NEXT(qual, filter);
82 
83 #undef NEXT
84 
85  // convert position and quality
86  char *endptr = NULL;
87  uint64 position = strtoll(pos, &endptr, 10);
88  if (!endptr || endptr == pos || *endptr != '\0')
89  {
90  log_error(stderr, "VCF file error (line %d): invalid position\n", line_counter);
91  return false;
92  }
93 
94  uint8 quality;
95  if (*qual == '.')
96  {
97  quality = 0xff;
98  } else {
99  quality = (uint8) strtol(qual, &endptr, 10);
100  if (!endptr || endptr == qual || *endptr != '\0')
101  {
102  log_warning(stderr, "VCF file error (line %d): invalid quality\n", line_counter);
103  quality = 0xff;
104  }
105  }
106 
107  // add an entry for each possible variant listed in this record
108  do {
109  char *next_base = strchr(alt, ',');
110  if (next_base)
111  *next_base = '\0';
112 
113  char *var;
114  // if this is a called monomorphic variant (i.e., a site which has been identified as always having the same allele)
115  // we store the reference string as the variant
116  if (strcmp(alt, ".") == 0)
117  var = ref;
118  else
119  var = alt;
120 
121  const uint32 ref_len = strlen(ref);
122  const uint32 var_len = strlen(var);
123 
124  SNP_sequence_index index(output.reference_sequences.size(), ref_len,
125  output.variants.size(), var_len);
126  output.ref_variant_index.push_back(index);
127 
128  output.reference_sequence_names.push_back(std::string(chrom));
129  output.positions.push_back(position);
130 
131  output.reference_sequences.resize(index.reference_start + ref_len);
132  string_to_dna16(ref, output.reference_sequences.begin() + index.reference_start);
133 
134  output.variants.resize(index.variant_start + var_len);
135  string_to_dna16(var, output.variants.begin() + index.variant_start);
136 
137  output.variant_qualities.push_back(quality);
138 
139  if (next_base)
140  alt = next_base + 1;
141  else
142  alt = NULL;
143  } while (alt && *alt != '\0');
144  }
145 
146  return true;
147 }
