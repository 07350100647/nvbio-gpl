51 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
52 void frame_opposite_mate(
53  const int policy,
54  const uint32 anchor,
55  const bool anchor_fw,
56  bool& left,
57  bool& fw)
58 {
59  const bool anchor_1 = (anchor == 0);
60 
61  switch(policy)
62  {
63  case io::PE_POLICY_FF:
64  {
65  left = (anchor_1 != anchor_fw);
66  fw = anchor_fw;
67  break;
68  }
69  case io::PE_POLICY_RR:
70  {
71  left = (anchor_1 == anchor_fw);
72  fw = anchor_fw;
73  break;
74  }
75  case io::PE_POLICY_FR:
76  {
77  left = !anchor_fw;
78  fw = !anchor_fw;
79  break;
80  }
81  case io::PE_POLICY_RF:
82  {
83  left = anchor_fw;
84  fw = !anchor_fw;
85  break;
86  }
87  }
88 }

93 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
94 int32 compute_target_score(const io::BestPairedAlignments& best, const int32 a_worst_score, const int32 o_worst_score)
95 {
96  if (best.has_second_paired() == false)
97  return a_worst_score + o_worst_score;
98 
99  // reproduce Bowtie2's score bounding behavior (under its default setting 'tighten = 3')
100  const int32 delta = best.best_score() - best.second_score();
101  return best.second_score() + delta * 3/4;
102 }

115 struct Backtracker
116 {
117  // constructor
118  NVBIO_FORCEINLINE NVBIO_DEVICE
119  Backtracker(vector vec, const uint32 _capacity) : out(vec), size(0), prev(255), capacity(_capacity) {}
120 
121  // encode a soft clipping operation
122  NVBIO_FORCEINLINE NVBIO_DEVICE
123  void clip(const uint32 l)
124  {
125  if (l)
126  {
127  NVBIO_CUDA_DEBUG_ASSERT( size < capacity, "Backtracker: exceeded maximum CIGAR size!\n" );
128  out[size++] = io::Cigar( io::Cigar::SOFT_CLIPPING, l );
129  }
130  }
131 
132  // encode a new operation
133  NVBIO_FORCEINLINE NVBIO_DEVICE
134  void push(uint8 type)
135  {
136  NVBIO_CUDA_DEBUG_ASSERT( type == aln::SUBSTITUTION ||
137  type == aln::DELETION ||
138  type == aln::INSERTION,
139  "Backtracker: unknown op type %u\n", uint32(type) );
140 
141  // perform run-length encoding of the symbol sequence
142  if (prev == type)
143  out[size-1u].m_len++;
144  else
145  {
146  NVBIO_CUDA_DEBUG_ASSERT( size < capacity, "Backtracker: exceeded maximum CIGAR size!\n" );
147  // encode a new symbol
148  out[size++] = io::Cigar( type, 1u );
149  prev = type;
150  }
151  }
152 
153  vector out;
154  uint32 size;
155  uint8 prev;
156  uint32 capacity;
157 };
