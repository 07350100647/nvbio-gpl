269 // initialize a set of BestAlignments
270 //
271 template <typename ReadBatch, typename ScoreFunction>
272 __global__
273 void init_alignments_kernel(
274  const ReadBatch read_batch,
275  const ScoreFunction worst_score_fun,
276  io::BestAlignments* best_data,
277  const uint32 mate)
278 {
279  const uint32 thread_id = threadIdx.x + BLOCKDIM*blockIdx.x;
280  if (thread_id >= read_batch.size()) return;
281 
282  // compute the read length
283  const uint2 read_range = read_batch.get_range( thread_id );
284  const uint32 read_len = read_range.y - read_range.x;
285 
286  const int32 worst_score = worst_score_fun( read_len );
287 
288  io::BestAlignments best;
289  best.m_a1 = io::Alignment( uint32(-1), io::Alignment::max_ed(), worst_score, mate );
290  best.m_a2 = io::Alignment( uint32(-1), io::Alignment::max_ed(), worst_score, mate );
291  best_data[ thread_id ] = best;
292 }

294 // initialize a set of BestAlignments
295 //
296 template <typename ReadBatch, typename ScoreFunction>
297 void init_alignments(
298  const ReadBatch read_batch,
299  const ScoreFunction worst_score_fun,
300  io::BestAlignments* best_data,
301  const uint32 mate = 0)
302 {
303  const int blocks = (read_batch.size() + BLOCKDIM-1) / BLOCKDIM;
