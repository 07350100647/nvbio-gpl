150 template <uint32 ALPHABET_SIZE_T>
151 struct trie_copy<ALPHABET_SIZE_T, CompressedTrie>
152 {
153  template <typename InTrieType, typename OutVectorType>
154  static void enact(
155  const InTrieType& in_trie,
156  const typename InTrieType::node_type in_node,
157  OutVectorType& out_nodes,
158  const uint32 out_node_index)
159  {
160  typedef typename InTrieType::node_type in_node_type;
161 
162  // check if the input node is a leaf
163  if (in_trie.is_leaf( in_node ))
164  {
165  out_nodes[ out_node_index ].set_size( in_trie.size( in_node ) );
166  return;
167  }
168 
169  Collector<ALPHABET_SIZE_T,in_node_type> children;
170 
171  in_trie.children( in_node, children );
172 
173  const uint32 child_offset = uint32( out_nodes.size() );
174  out_nodes.resize( child_offset + children.count );
175 
176  out_nodes[ out_node_index ] = TrieNode<CompressedTrie>(
177  child_offset,
178  children.mask );
179 
180  for (uint32 i = 0; i < children.count; ++i)
181  {
182  enact(
183  in_trie,
184  children.nodes[i],
185  out_nodes,
186  child_offset + i );
187  }
188  }
189 };
