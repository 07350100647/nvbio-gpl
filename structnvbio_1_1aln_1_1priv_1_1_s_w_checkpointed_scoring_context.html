template< uint32 BAND_LEN, AlignmentType TYPE, typename checkpoint_type>
struct nvbio::aln::priv::SWCheckpointedScoringContext< BAND_LEN, TYPE, checkpoint_type >

A helper checkpointed-scoring context class for sw_alignment_score_dispatch which allows to perform scoring in multiple passes, saving & restoring a checkpoint each time.
