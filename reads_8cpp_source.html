34  const QualityEncoding qualities,
35  const uint32 max_reads,
36  const uint32 truncate_read_len)
37 {
38  // parse out file extension; look for .fastq.gz, .fastq suffixes
39  uint32 len = uint32( strlen(read_file_name) );
40  bool is_gzipped = false;
41 
42  // do we have a .gz suffix?
43  if (len >= strlen(".gz"))
44  {
45  if (strcmp(&read_file_name[len - strlen(".gz")], ".gz") == 0)
46  {
47  is_gzipped = true;
48  len = uint32(len - strlen(".gz"));
49  }
50  }
51 
52  // check for fastq suffix
53  if (len >= strlen(".fastq"))
54  {
55  if (strncmp(&read_file_name[len - strlen(".fastq")], ".fastq", strlen(".fastq")) == 0)
56  {
57  return new ReadDataFile_FASTQ_gz(read_file_name,
58  qualities,
59  max_reads,
60  truncate_read_len);
61  }
62  }
63 
64  if (len >= strlen(".fq"))
65  {
66  if (strncmp(&read_file_name[len - strlen(".fq")], ".fq", strlen(".fq")) == 0)
67  {
68  return new ReadDataFile_FASTQ_gz(read_file_name,
69  qualities,
70  max_reads,
71  truncate_read_len);
72  }
73  }
74 
75  // check for sam suffix
76  if (len >= strlen(".sam"))
77  {
78  if (strncmp(&read_file_name[len - strlen(".sam")], ".sam", strlen(".sam")) == 0)
79  {
80  ReadDataFile_SAM *ret;
81

93  }
94  }
95 
96  // check for bam suffix
97  if (len >= strlen(".bam"))
98  {
99  if (strncmp(&read_file_name[len - strlen(".bam")], ".bam", strlen(".bam")) == 0)
100  {
101  ReadDataFile_BAM *ret;
102 
103  ret = new ReadDataFile_BAM(read_file_name,
104  max_reads,

106

108  {
109  delete ret;

111  }
112 
113  return ret;
114  }
115  }
116 
117  // we don't actually know what this is; guess fastq
118  log_warning(stderr, "could not determine file type for %s; guessing %sfastq\n", read_file_name, is_gzipped ? "compressed " : "");
119  return new ReadDataFile_FASTQ_gz(read_file_name,

274 // add a read to this batch
275 void ReadDataRAM::push_back(uint32 read_len,
276  const char *name,
277  const uint8* read,
278  const uint8* quality,
279  QualityEncoding q_encoding,
280  uint32 truncate_read_len,
281  uint32 conversion_flags)
282 {
283  // truncate read
284  // xxx: should we do this silently?
285  read_len = nvbio::min(read_len, truncate_read_len);
286 
287  assert(read_len);
288 
289  // resize the reads & quality buffers
290  {
291  const uint32 bps_per_word = 32 / ReadData::READ_BITS;
292  const uint32 words = (m_read_stream_len + read_len + bps_per_word - 1) / bps_per_word;
293  m_read_vec.resize(words);
294  m_qual_vec.resize(m_read_stream_len + read_len);
295  m_read_stream_words = words;
296  }
297 
298  // encode the read data
299  ReadData::read_stream_type stream(&m_read_vec[0]);
300  for(uint32 i = 0; i < read_len; i++)
301  {
302  char bp = read[i];
303 
304  if (conversion_flags & COMPLEMENT)
305  {
306  bp = complement_bp(bp);
307  }
308 
309  // xxx: note that we're pushing in reverse order by default
310  // this is to be consistent with reads_fastq.cpp
311  if (conversion_flags & REVERSE)
312  {
313  stream[m_read_stream_len + i] = nst_nt4_encode(bp);
314  m_qual_vec[m_read_stream_len + i] = convert_to_phred_quality(q_encoding, quality[i]);
315  } else {
316  stream[m_read_stream_len + read_len - i - 1] = nst_nt4_encode(bp);
317  m_qual_vec[m_read_stream_len + read_len - i - 1] = convert_to_phred_quality(q_encoding, quality[i]);
318  }
319  }
320 
321  // update read and bp counts
322  m_n_reads++;
323  m_read_stream_len += read_len;
324  m_read_index_vec.push_back(m_read_stream_len);
325 
326  m_min_read_len = nvbio::min(m_min_read_len, read_len);
327  m_max_read_len = nvbio::max(m_max_read_len, read_len);
328 
329  // store the read name
330  const uint32 name_len = uint32(strlen(name));
331  const uint32 name_offset = m_name_stream_len;
332 
333  m_name_vec.resize(name_offset + name_len + 1);
334  strcpy(&m_name_vec[name_offset], name);
335 
336  m_name_stream_len += name_len + 1;
337  m_name_index_vec.push_back(m_name_stream_len);
338 }

340 // utility function to alloc and copy a vector in device memory
341 template <typename T>
342 static void cudaAllocAndCopyVector(T*& dst, const T* src, const uint32 words, uint64& allocated)
343 {
344  const uint32 words4 = 4u * ((words + 3u) / 4u);
345  if (src)
346  {
347  cudaMalloc( &dst, sizeof(T) * words4 );
348  if (dst == NULL)
349  throw std::bad_alloc(WINONLY("ReadDataCUDA: not enough device memory"));
350 
351  cudaMemcpy( dst, src, sizeof(T) * words, cudaMemcpyHostToDevice );
352 
353  allocated += words4 * sizeof(T);
354  }
355  else
356  dst = NULL;
357 }

402 // grab the next batch of reads into a host memory buffer
403 ReadData *ReadDataFile::next(const uint32 batch_size)
404 {
405  const uint32 to_load = std::min(m_max_reads - m_loaded, batch_size);
406  uint32 m;
407 
408  if (!is_ok() || to_load == 0)
409  return NULL;
410 
411  ReadDataRAM *reads = new ReadDataRAM();
412 
413  m = 0;
414  while (m < to_load)
415  {
416  // load 100 at a time if possible
417  const uint32 q = std::min(to_load - m, uint32(100));
418  int n;
419 
420  n = nextChunk(reads, q);
421  assert(n <= (int) q);
422  if (n == 0)
423  {
424  break;
425  }
426 
427  m += n;
428  assert(m <= to_load);
429  }
430 
431  if (m == 0)
432  return NULL;
433 
434  m_loaded += m;
435 
436  reads->end_batch();
437 
438  return reads;
439 }
