403 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
404 uint32 extract_word_generic(
405  const string_type string,
406  const index_type string_len,
407  const index_type suffix_idx,
408  const uint32 w)
409 {
410  const uint32 SYMBOLS_PER_WORD = uint32(WORD_BITS - DOLLAR_BITS)/SYMBOL_SIZE;
411  const uint32 SYMBOL_OFFSET = uint32(WORD_BITS) - SYMBOL_SIZE;
412 
413  uint32 word = 0u;
414  for (uint32 j = 0; j < SYMBOLS_PER_WORD; ++j)
415  {
416  const index_type jj = suffix_idx + w*SYMBOLS_PER_WORD + j;
417  const uint32 c = jj < string_len ? string[jj] : 0u;
418  word |= (c << (SYMBOL_OFFSET - j*SYMBOL_SIZE));
419  }
420 
421  if (DOLLAR_BITS)
422  {
423  // encode the dollar's position in the least significant bits of the word
424  const uint32 dollar_offset =
425  string_len <= suffix_idx + w*SYMBOLS_PER_WORD + SYMBOLS_PER_WORD ? // is there a dollar sign?
426  (string_len < suffix_idx + w*SYMBOLS_PER_WORD) ? 0u :
427  uint32(string_len - suffix_idx - w*SYMBOLS_PER_WORD) :
428  (1u << DOLLAR_BITS)-1u; // no dollar sign in this word
429 
430  return word | dollar_offset;
431  }
432  else
433  return word;
434 }

446 template <uint32 WORD_BITS, uint32 DOLLAR_BITS, uint32 SYMBOL_SIZE, typename storage_type, typename index_type, typename sufindex_type>
447 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
448 typename std::iterator_traits<storage_type>::value_type extract_word_packed(
449  const storage_type base_words,
450  const index_type string_len,
451  const index_type string_off,
452  const sufindex_type suffix_idx,
453  const uint32 w)
454 {
455  typedef typename std::iterator_traits<storage_type>::value_type word_type;
456 
457  const uint32 STORAGE_BITS = uint32( 8u * sizeof(word_type) );
458  const uint32 STORAGE_SYMBOLS = STORAGE_BITS / SYMBOL_SIZE;
459  const uint32 SYMBOLS_PER_WORD = uint32(WORD_BITS - DOLLAR_BITS)/SYMBOL_SIZE;
460  //const uint32 SYMBOL_OFFSET = uint32(WORD_BITS) - SYMBOL_SIZE;
461 
462  const sufindex_type suffix_off = suffix_idx + w*SYMBOLS_PER_WORD; // suffix offset
463 
464  // do we have any symbols to encode?
465  if (suffix_off >= string_len)
466  return 0u;
467 
468  const index_type range_len = string_len - suffix_off; // partial suffix length
469  const index_type range_off = string_off + suffix_off; // partial suffix offset
470 
471  const uint32 n_symbols = (uint32)nvbio::min( range_len, index_type(SYMBOLS_PER_WORD) ); // symbols to pack
472 
473  //
474  // As SYMBOLS_PER_WORD is less than 32, we know that the symbols we are looking for
475  // will span at most 2 32-bit words.
476  // Of the n_symbols we want to read, there might be n1 in the first word, and n2 in the
477  // second word.
478  // As the beginning of our symbol stream (range_off) might stride the 32-bit word boundary,
479  // the highest m1 = range_off % 16 symbols of the first word might have to be discarded,
480  // and we'll find our n1 symbols in the following position:
481  //
482  // |-------------------------------------------|
483  // |* * * * * *| x x x x x x x x x | * * * * * |
484  // |-------------------------------------------|
485  // | m1 | n1 | r1 |
486  //
487  // What we do is shifting the n1 symbols to the top of the 32-bit word (i.e. m1 to the left).
488  // Clearing the remaining symbols is only needed if n1 == n_symbols; if n1 < n_symbols, r1 will
489  // be necessarily zero.
490  //
491  // At this point, we might have n2 more symbols to read in the highest bits of the second word:
492  //
493  // |-------------------------------------------|
494  // | y y y y y y y y | * * * * * * * * * * * * |
495  // |-------------------------------------------|
496  // | n2 | r2 |
497  //
498  // which we need to shift right by (n1*SYMBOL_SIZE) bits.
499  // At the very end, we'll shift everything right by (32 - WORD_BITS) bits in order to have
500  // our output tightly packed in the lowest WORD_BITS:
501  //
502  // 32 WORD_BITS DOLLAR_BITS 0
503  // |-----------|-----------------------|-------|
504  // | * * * * * | x x x x | y y y | 0 0 | $ $ $ | // notice the possible presence of 0's before
505  // |-------------------------------------------| // the $ sign: these are bits that need to be
506  // | | n1 | n2 | | | // cleared if the suffix is short
507  //
508 
509  const uint32 k1 = uint32( range_off/STORAGE_SYMBOLS ); // index of the first word
510 
511  const uint32 m1 = range_off & (STORAGE_SYMBOLS-1); // offset in the word
512  const uint32 r1 = STORAGE_SYMBOLS - m1; // symbols to read
513  const word_type word1 = (base_words[ k1 ] << (m1*SYMBOL_SIZE)); // fetch the first word, shifted left
514 
515  word_type word = word1;
516 
517  if (n_symbols > r1) // do we need to read another word?
518  {
519  const word_type word2 = base_words[ k1+1u ]; // fetch the second word
520  word |= word2 >> (r1*SYMBOL_SIZE); // shift by n1 symbols to the right
521  }
522 
523  word >>= (STORAGE_BITS - WORD_BITS); // align the top to WORD_BITS
524 
525  // clear every symbol we don't need among the word's LSD
526  word &= clearmask<word_type>( WORD_BITS - n_symbols*SYMBOL_SIZE );
527 
528  if (DOLLAR_BITS)
529  {
530  // encode the dollar's position in the least significant bits of the word
531  const word_type dollar_offset =
532  range_len <= SYMBOLS_PER_WORD ? // is there a dollar sign?
533  range_len :
534  (1u << DOLLAR_BITS)-1u; // no dollar sign in this word
535 
536  return word | dollar_offset;
537  }
538  else
539  return word;
540 }
