264 // enact the batch execution
265 //
266 template <typename stream_type>
267 void BatchedAlignmentScore<stream_type,ThreadParallelScheduler>::enact(stream_type stream, uint64 temp_size, uint8* temp)
268 {
269  const uint32 column_size = equal<typename aligner_type::algorithm_tag,PatternBlockingTag>() ?
270  uint32( stream.max_pattern_length() ) :
271  uint32( stream.max_text_length() );
272 
273  // if the column is small, let's just use statically allocated local memory
274  if (column_size <= 1024)
275  {
276  const uint32 n_blocks = (stream.size() + BLOCKDIM-1) / BLOCKDIM;
277 
278  lmem_batched_alignment_score_kernel<BLOCKDIM,1024> <<<n_blocks, BLOCKDIM>>>(
279  stream,
280  (cell_type*)temp,
281  stream.size() );
282  }
283  else
284  {
285  // the column is large to be allocated in local memory, let's use global memory
286  const uint64 min_temp_size = min_temp_storage(
287  stream.max_pattern_length(),
288  stream.max_text_length(),
289  stream.size() );
290 
291  thrust::device_vector<uint8> temp_dvec;
292  if (temp == NULL)
293  {
294  temp_size = nvbio::max( min_temp_size, temp_size );
295  temp_dvec.resize( temp_size );
296  temp = nvbio::device_view( temp_dvec );
297  }
298 
299  // set the queue capacity based on available memory
300  const uint32 queue_capacity = uint32( temp_size / column_storage( stream.max_pattern_length(), stream.max_text_length() ) );
301 
302  if (queue_capacity >= stream.size())
303  {
304  const uint32 n_blocks = (stream.size() + BLOCKDIM-1) / BLOCKDIM;
305 
306  batched_alignment_score_kernel<BLOCKDIM> <<<n_blocks, BLOCKDIM>>>(
307  stream,
308  (cell_type*)temp,
309  stream.size() );
310  }
311  else
312  {
313  // compute the number of blocks we are going to launch
314  const uint32 n_blocks = nvbio::max( nvbio::min(
315  (uint32)cuda::max_active_blocks( persistent_batched_alignment_score_kernel<BLOCKDIM,stream_type,cell_type>, BLOCKDIM, 0u ),
316  queue_capacity / BLOCKDIM ), 1u );
317 
318  persistent_batched_alignment_score_kernel<BLOCKDIM> <<<n_blocks, BLOCKDIM>>>(
319  stream,
320  (cell_type*)temp,
321  queue_capacity );
322  }
323  }
324 }

370 template <typename stream_type>
371 void BatchedAlignmentScore<stream_type,WarpParallelScheduler>::enact(stream_type stream, uint64 temp_size, uint8* temp)
372 {
373  const uint64 min_temp_size = min_temp_storage(
374  stream.max_pattern_length(),
375  stream.max_text_length(),
376  stream.size() );
377 
378  thrust::device_vector<uint8> temp_dvec;
379  if (temp == NULL)
380  {
381  temp_size = nvbio::max( min_temp_size, temp_size );
382  temp_dvec.resize( temp_size );
383  temp = nvbio::device_view( temp_dvec );
384  }
385 
386  NVBIO_VAR_UNUSED static const uint32 WARP_SIZE = cuda::Arch::WARP_SIZE;
387 
388  // set the queue capacity based on available memory
389  const uint32 queue_capacity = align_down<WARP_SIZE>( uint32( temp_size / (align<WARP_SIZE>( stream.max_text_length() ) * sizeof(cell_type)) ) );
390 
391  const uint32 BLOCKWARPS = BLOCKDIM >> cuda::Arch::LOG_WARP_SIZE;
392  if (queue_capacity >= stream.size())
393  {
394  const uint32 n_warps = stream.size();
395  const uint32 n_blocks = (n_warps + BLOCKWARPS-1) / BLOCKWARPS;
396 
397  warp_batched_alignment_score_kernel<BLOCKDIM> <<<n_blocks, BLOCKDIM>>>(
398  stream,
399  (cell_type*)temp,
400  align<WARP_SIZE>( stream.size() ) ); // make sure everything is properly aligned
401  }
402  else
403  {
404  // compute the number of blocks we are going to launch
405  const uint32 n_blocks = nvbio::max( nvbio::min(
406  (uint32)cuda::max_active_blocks( warp_persistent_batched_alignment_score_kernel<BLOCKDIM,stream_type,cell_type>, BLOCKDIM, 0u ),
407  queue_capacity / BLOCKDIM ), 1u );
408 
409  warp_persistent_batched_alignment_score_kernel<BLOCKDIM> <<<n_blocks, BLOCKDIM>>>(
410  stream,
411  (cell_type*)temp,
412  queue_capacity );
413  }
414 }

678 // enact the batch execution
679 //
680 template <uint32 CHECKPOINTS, typename stream_type>
681 void BatchedAlignmentTraceback<CHECKPOINTS,stream_type,ThreadParallelScheduler>::enact(stream_type stream, uint64 temp_size, uint8* temp)
682 {
683  const uint64 min_temp_size = min_temp_storage(
684  stream.max_pattern_length(),
685  stream.max_text_length(),
686  stream.size() );
687 
688  thrust::device_vector<uint8> temp_dvec;
689  if (temp == NULL)
690  {
691  temp_size = nvbio::max( min_temp_size, temp_size );
692  temp_dvec.resize( temp_size );
693  temp = nvbio::device_view( temp_dvec );
694  }
695 
696  // set the queue capacity based on available memory
697  const uint32 max_pattern_len = stream.max_pattern_length();
698  const uint32 max_text_len = stream.max_text_length();
699  const uint32 queue_capacity = uint32( temp_size / element_storage( max_pattern_len, max_text_len ) );
700 
701  const uint64 column_size = column_storage( max_pattern_len, max_text_len );
702  const uint64 checkpoints_size = checkpoint_storage( max_pattern_len, max_text_len );
703 
704  if (queue_capacity >= stream.size())
705  {
706  const uint32 n_blocks = (stream.size() + BLOCKDIM-1) / BLOCKDIM;
707 
708  cell_type* checkpoints = (cell_type*)(temp);
709  cell_type* columns = (cell_type*)(temp + (checkpoints_size) * stream.size());
710  uint32* submatrices = (uint32*) (temp + (checkpoints_size + column_size) * stream.size());
711 
712  batched_alignment_traceback_kernel<BLOCKDIM,CHECKPOINTS> <<<n_blocks, BLOCKDIM>>>(
713  stream,
714  checkpoints,
715  submatrices,
716  columns,
717  stream.size() );
718  }
719  else
720  {
721  // compute the number of blocks we are going to launch
722  const uint32 n_blocks = nvbio::max( nvbio::min(
723  (uint32)cuda::max_active_blocks( persistent_batched_alignment_traceback_kernel<BLOCKDIM,CHECKPOINTS,stream_type,cell_type>, BLOCKDIM, 0u ),
724  queue_capacity / BLOCKDIM ), 1u );
725 
726  cell_type* checkpoints = (cell_type*)(temp);
727  cell_type* columns = (cell_type*)(temp + (checkpoints_size) * queue_capacity);
728  uint32* submatrices = (uint32*) (temp + (checkpoints_size + column_size) * queue_capacity);
729 
730  persistent_batched_alignment_traceback_kernel<BLOCKDIM,CHECKPOINTS> <<<n_blocks, BLOCKDIM>>>(
731  stream,
732  checkpoints,
733  submatrices,
734  columns,
735  queue_capacity );
736  }
