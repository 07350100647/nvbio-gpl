96 struct DeviceVectorArrayView
97 {
100  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
101  DeviceVectorArrayView(
102  T* arena = NULL,
103  uint32* index = NULL,
104  uint32* pool = NULL,
105  uint32 size = 0u)
106  : m_arena(arena), m_index(index), m_pool(pool), m_size(size) {}
107 
108 #ifdef __CUDACC__
109  NVBIO_FORCEINLINE NVBIO_DEVICE
112  T* alloc(const uint32 index, const uint32 size)
113  {
114  const uint32 slot = atomicAdd( m_pool, size );
115  if (slot + size >= m_size)
116  {
117  // mark an out-of-bounds allocation
118  m_index[index] = m_size;
119  return NULL;
120  }
121  m_index[index] = slot;
122  return m_arena + slot;
123  }
124 #endif
125 
128  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
129  T* operator[](const uint32 index) const
130  {
131  // for unsuccessful allocations m_index is set to m_size - in that case we return NULL
132  return (m_index[index] < m_size) ? m_arena + m_index[index] : NULL;
133  }
134 
137  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
138  uint32 slot(const uint32 index) const { return m_index[index]; }
139
