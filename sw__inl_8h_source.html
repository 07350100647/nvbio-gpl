342 struct sw_alignment_score_dispatch
343 {
344  template <
345  bool CHECK_M,
346  typename context_type,
347  typename query_cache,
348  typename score_type,
349  typename temp_iterator,
350  typename sink_type>
351  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
352  static void update_row(
353  context_type& context,
354  const uint32 block,
355  const uint32 M,
356  const uint32 i,
357  const uint32 N,
358  const symbol_type r_i,
359  const query_cache q_cache,
360  temp_iterator temp,
361  score_type& temp_i,
362  score_type* band,
363  sink_type& sink,
364  score_type& max_score,
365  const score_type V,
366  const score_type S,
367  const score_type G,
368  const score_type I,
369  const score_type zero,
370  const symbol_type invalid_symbol)
371  {
372  // set the 0-th coefficient in the band to be equal to the (i-1)-th row of the left column (diagonal term)
373  score_type prev = temp_i;
374 
375  // set the 0-th coefficient in the band to be equal to the i-th row of the left column (left term)
376  band[0] = temp_i = temp[i];
377 
378 #if 0
379  // do top-diagonal cell updates
380  #pragma unroll
381  for (uint32 j = 1; j <= BAND_LEN; ++j)
382  {
383  const symbol_type q_j = q_cache[ j-1 ];
384  const score_type S_ij = (r_i == q_j) ? V : S;
385  const score_type diagonal = prev + S_ij;
386  const score_type top = band[j] + G;
387  prev = band[j];
388  band[j] = nvbio::max( top, diagonal );
389  }
390 
391  // do left cell updates
392  #pragma unroll
393  for (uint32 j = 1; j <= BAND_LEN; ++j)
394  {
395  const score_type left = band[j-1] + I;
396  score_type hi = nvbio::max( band[j], left );
397  hi = TYPE == LOCAL ? nvbio::max( hi, zero ) : hi; // clamp to zero
398  band[j] = hi;
399  }
400 #else
401  // update all cells
402  #pragma unroll
403  for (uint32 j = 1; j <= BAND_LEN; ++j)
404  {
405  const symbol_type q_j = q_cache[ j-1 ];
406  const score_type S_ij = (r_i == q_j) ? V : S;
407  const score_type diagonal = prev + S_ij;
408  const score_type top = band[j] + G;
409  const score_type left = band[j-1] + I;
410  score_type hi = nvbio::max3( top, left, diagonal );
411  hi = TYPE == LOCAL ? nvbio::max( hi, zero ) : hi; // clamp to zero
412  prev = band[j];
413  band[j] = hi;
414 
415  context.new_cell(
416  i, N,
417  block + j - 1, M,
418  hi,
419  top > left ?
420  (top > diagonal ? DELETION : SUBSTITUTION) :
421  (left > diagonal ? INSERTION : SUBSTITUTION) );
422  }
423 #endif
424 
425  // save the last entry of the band
426  temp[i] = band[ BAND_LEN ];
427 
428  max_score = nvbio::max( max_score, band[ BAND_LEN ] );
429 
430  if (TYPE == LOCAL)
431  {
432  if (CHECK_M)
433  {
434  // during local alignment we save the best score across all bands
435  for (uint32 j = 1; j <= BAND_LEN; ++j)
436  {
437  if (block + j <= M)
438  sink.report( band[j], make_uint2( i+1, block+j ) );
439  }
440  }
441  else
442  {
443  // during local alignment we save the best score across all bands
444  for (uint32 j = 1; j <= BAND_LEN; ++j)
445  sink.report( band[j], make_uint2( i+1, block+j ) );
446  }
447  }
448  else if (CHECK_M)
449  {
450  if (TYPE == SEMI_GLOBAL)
451  {
452  // during semi-global alignment we save the best score across the last column H[*][M], at each row
453  save_boundary<BAND_LEN>( block, M, band, i, sink );
454  }
455  }

530  template <
531  typename context_type,
532  typename string_type,
533  typename qual_type,
534  typename ref_type,
535  typename scoring_type,
536  typename sink_type,
537  typename column_type>
538  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
539  static
540  bool run(
541  const scoring_type& scoring,
542  context_type& context,
543  string_type query,
544  qual_type quals,
545  ref_type ref,
546  const int32 min_score,
547  sink_type& sink,
548  const uint32 window_begin,
549  const uint32 window_end,
550  column_type temp)
551  {
552  const uint32 M = query.length();
553  const uint32 N = ref.length();
554 
555  typedef int32 score_type;
556  score_type band[BAND_LEN+1];
557 
558  const score_type G = scoring.deletion();
559  const score_type I = scoring.insertion();
560  const score_type S = scoring.mismatch();
561  const score_type V = scoring.match();
562  const score_type zero = score_type(0);
563 
564  // we are going to process the matrix in stripes, where each partial row within each stripe is processed
565  // in registers: hence, we need to save the right-boundary of each stripe (a full matrix column) in some
566  // temporary storage.
567 
568  // fill row 0 with zeros, this will never be updated
569  uint8 q_cache[ BAND_LEN ];
570 
571  const uint8 invalid_symbol = 255u;
572 
573  // initialize the first column
574  context.init( window_begin, N, temp, scoring, zero );
575 
576  const uint32 end_block = (window_end == M) ?
577  nvbio::max( BAND_LEN, BAND_LEN * ((M + BAND_LEN-1) / BAND_LEN) ) :
578  window_end + BAND_LEN;
579 
580  // loop across the short edge of the DP matrix (i.e. the columns)
581  for (uint32 block = window_begin; block + BAND_LEN < end_block; block += BAND_LEN)
582  {
583  // save the previous column
584  context.previous_column( block, N, temp );
585 
586  // initialize the first band (corresponding to the 0-th row of the DP matrix)
587  #pragma unroll
588  for (uint32 j = 0; j <= BAND_LEN; ++j)
589  band[j] = (TYPE != LOCAL) ? G * (block + j) : zero;
590 
591  // load a block of entries from each query
592  #pragma unroll
593  for (uint32 t = 0; t < BAND_LEN; ++t)
594  q_cache[ t ] = query[ block + t ];
595 
596  score_type max_score = Field_traits<score_type>::min();
597 
598  score_type temp_i = band[0];
599 
600  // loop across the long edge of the DP matrix (i.e. the rows)
601  for (uint32 i = 0; i < N; ++i)
602  {
603  // load the new character from the reference
604  const uint8 r_i = ref[i];
605 
606  update_row<false>(
607  context,
608  block, M,
609  i, N,
610  r_i,
611  q_cache,
612  temp,
613  temp_i,
614  band,
615  sink,
616  max_score,
617  V,S,G,I,
618  zero,
619  invalid_symbol );
620  }
621 
622  // we are now (M - block - BAND_LEN) columns from the last one: check whether
623  // we could theoretically reach the minimums core
624  const score_type missing_cols = score_type(M - block - BAND_LEN);
625  if (max_score + missing_cols * scoring.match(255) < score_type( min_score ))
626  return false;
627  }
628 
629  if (window_end == M)
630  {
631  const uint32 block = end_block - BAND_LEN;
632 
633  // save the previous column
634  context.previous_column( block, N, temp );
635 
636  // initialize the first band (corresponding to the 0-th row of the DP matrix)
637  for (uint32 j = 0; j <= BAND_LEN; ++j)
638  band[j] = (TYPE != LOCAL) ? G * (block + j) : zero;
639 
640  // load a block of entries from each query
641  const uint32 block_end = nvbio::min( block + BAND_LEN, M );
642  #pragma unroll
643  for (uint32 t = 0; t < BAND_LEN; ++t)
644  {
645  if (block + t < block_end)
646  q_cache[ t ] = query[ block + t ];
647  }
648 
649  score_type max_score = Field_traits<score_type>::min();
650 
651  score_type temp_i = band[0];
652 
653  // loop across the long edge of the DP matrix (i.e. the rows)
654  for (uint32 i = 0; i < N; ++i)
655  {
656  // load the new character from the reference
657  const uint8 r_i = ref[i];
658 
659  update_row<true>(
660  context,
661  block, M,
662  i, N,
663  r_i,
664  q_cache,
665  temp,
666  temp_i,
667  band,
668  sink,
669  max_score,
670  V,S,G,I,
671  zero,
672  invalid_symbol );
673  }
674  }
675 
676  if (TYPE == GLOBAL)
677  save_Mth<BAND_LEN>( M, band, N-1, sink );
678 
679  return true;
680  }
