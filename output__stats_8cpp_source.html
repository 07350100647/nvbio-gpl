25 // keep track of alignment statistics for a given alignment

28 // be the opposite mate
29 void IOStats::track_alignment_statistics(const AlignmentData& alignment,
30  const AlignmentData& mate,
31  const uint8 mapq)
32 {
33  n_reads++;
34 
35  // keep track of mapping quality histogram
36  mapq_bins[mapq]++;
37 
38  if (!alignment.best->is_aligned())
39  {
40  // sanity check
41  NVBIO_CUDA_ASSERT(alignment.second_best->is_mapped() == false);
42 
43  // if first anchor was not mapped, nothing was; count as unmapped
44  mapped_ed_correlation[0][0]++;
45  return;
46  }
47 
48  // count this read as mapped
49  n_mapped++;
50 
51  if (!alignment.second_best->is_aligned())
52  {
53  // we only have one score; count as a unique alignment
54  n_unique++;
55  } else {
56  // we have two best scores, which implies two (or more) alignment positions
57  // count as multiple alignment
58  n_multiple++;
59 
60  // compute final alignment score
61  int32 first = alignment.best->score();
62  int32 second = alignment.second_best->score();
63 
64  if (mate.valid)
65  {
66  first += mate.best->score();
67  second += mate.second_best->score();
68  }
69 
70  // if the two best scores are equal, count as ambiguous
71  if (first == second)
72  n_ambiguous++;
73  else {
74  // else, the first score must be higher...
75  NVBIO_CUDA_ASSERT(first > second);
77  n_unambiguous++;
78  }
79 
80  // compute edit distance scores
81  uint32 first_ed = alignment.best->ed();
82  uint32 second_ed = alignment.second_best->score();
83 
84  // update best edit-distance histograms
85  if (first_ed < mapped_ed_histogram.size())
86  {
87  mapped_ed_histogram[first_ed]++;
88  if (alignment.best->m_rc)
89  {
90  mapped_ed_histogram_fwd[first_ed]++;
91  } else {
92  mapped_ed_histogram_rev[first_ed]++;
93  }
94  }
95 
96  // track edit-distance correlation
97  if (first_ed + 1 < 64)
98  {
99  if (second_ed + 1 < 64)
100  {
101  mapped_ed_correlation[first_ed + 1][second_ed + 1]++;
102  } else {
103  mapped_ed_correlation[first_ed + 1][0]++;
104  }
105  }
106  }
