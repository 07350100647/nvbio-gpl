111 template <uint32 SYMBOL_SIZE, bool BIG_ENDIAN, typename word_type>
112 struct HostBWTHandler< PackedStreamIterator< PackedStream<word_type*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> > > : public BaseBWTHandler
113 {
114  typedef PackedStreamIterator< PackedStream<word_type*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> > OutputIterator;
115 
116  static const uint32 WORD_SIZE = uint32( 8u * sizeof(word_type) );
117  static const uint32 SYMBOLS_PER_WORD = WORD_SIZE / SYMBOL_SIZE;
118 
119 
122  HostBWTHandler(OutputIterator _output) : output(_output), offset(0) {}
123 
126  void process(
127  const uint32 n_suffixes,
128  const uint8* h_bwt,
129  const uint8* d_bwt,
130  const uint2* h_suffixes,
131  const uint2* d_suffixes,
132  const uint32* d_indices)
133  {
134  const uint32 word_offset = offset & (SYMBOLS_PER_WORD-1);
135  uint32 word_rem = 0;
136  uint32 word_idx = offset / SYMBOLS_PER_WORD;
137 
138  word_type* words = output.container().stream();
139 
140  if (word_offset)
141  {
142  // compute how many symbols we still need to encode to fill the current word
143  word_rem = SYMBOLS_PER_WORD - word_offset;
144 
145  // fetch the word in question
146  word_type word = words[ word_idx ];
147 
148  for (uint32 i = 0; i < word_rem; ++i)
149  {
150  const uint32 bit_idx = (word_offset + i) * SYMBOL_SIZE;
151  const uint32 symbol_offset = BIG_ENDIAN ? (WORD_SIZE - SYMBOL_SIZE - bit_idx) : bit_idx;
152  const word_type symbol = word_type(h_bwt[i]) << symbol_offset;
153 
154  // set bits
155  word |= symbol;
156  }
157 
158  // write out the word
159  words[ word_idx ] = word;
160 
161  // advance word_idx
162  ++word_idx;
163  }
164 
165  for (uint32 i = word_rem; i < n_suffixes; i += SYMBOLS_PER_WORD)
166  {
167  // encode a word's worth of characters
168  word_type word = 0u;
169 
170  const uint32 n_symbols = nvbio::min( SYMBOLS_PER_WORD, n_suffixes - i );
171 
172  for (uint32 j = 0; j < n_symbols; ++j)
173  {
174  const uint32 bit_idx = j * SYMBOL_SIZE;
175  const uint32 symbol_offset = BIG_ENDIAN ? (WORD_SIZE - SYMBOL_SIZE - bit_idx) : bit_idx;
176  const word_type symbol = word_type(h_bwt[i + j]) << symbol_offset;
177 
178  // set bits
179  word |= symbol;
180  }
181 
182  // write out the word and advance word_idx
183  words[ ++word_idx ] = word;
184  }
185 
186  // advance the offset
187  offset += n_suffixes;
188  }
189 
190  OutputIterator output;
191  uint64 offset;
192 };
